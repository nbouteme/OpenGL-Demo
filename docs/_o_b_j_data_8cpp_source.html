87 OBJData::OBJData(const string &objFile)
88 {
89  auto it = objFile.begin();
90  string kTokens = "\n#sovnt/f";
91 
92  while(it != objFile.end()) // Dans cette boucle, on "tokenize" le fichier: on le decoupe de maniere a bien separer ce qui nous interresse (a.k.a. les elements qui font parti du format de fichier) et ce qui nous interesse moins (espaces)
93  {
94  string token;
95 
96  while(*it != ' ' && kTokens.find(*it) == string::npos)
97  {
98  token += *it;
99  ++it;
100  }
101  if(!token.empty())
102  m_tokens.push_back(token);
103  if(kTokens.find(*it) != string::npos)
104  m_tokens.push_back(string(1, *it));
105 
106  ++it;
107  }
108 
109  for(auto it = m_tokens.begin(); it < m_tokens.end();) // C'est ici que se fait le parsing: on transforme les donnees
110  { // dans un format utilisable par les fonctions OpenGL
111  switch(it[0][0])
112  {
113  case 'f': // un f veut dire qu'apres, on a une face
114  m_faces.push_back(parseFace(it));
115  break;
116  case 'v': // le reste de la boucle suit la meme logique
117  switch(it[1][0])
118  {
119  case 'n':
120  m_normals.push_back(parseNormal(++it));
121  break;
122  case 't':
123  m_text.push_back(parseTexUV(it));
124  break;
125  default:
126  m_vertices.push_back(parseVert(it));
127  }
128  case 'o': // nom: ignorer
129  case '#': // commentaire
130  case 's': // groupe de smooth vertices, aka. on s'en fout
131  break;
132  }
133  while(it < m_tokens.end() && *it != "\n") ++it; // Si on rencontre un token inutile, on avance jusqu'au prochain
134  while(it < m_tokens.end() && *it == "\n") ++it;
135  }
136 }
