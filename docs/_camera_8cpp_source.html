16 void onHovering(GLFWwindow *win, double x, double y)
17 {
18  (void)win;
19  static int oldx, oldy;
20  // si le bouton de la souris est enfoncé
21  if(glfwGetMouseButton(*(GLWindow*)Application::getSingleton()->getWindow().get(), 0) == GLFW_PRESS)
22  {
23  // Recupere la caméra
24  Camera &self = *((GLWindow*)Application::getSingleton()->getWindow().get())->m_cam;
25 
26  // calcule la difference d'angle par pixel pour un champs de vision de 45 deg
27  float xangle = 3.141529f / 4 / self.width;
28  float yangle = 3.141529f / 4 / self.height;
29 
30  vec3 right = cross(vec3(0.0f, 1.0f, 0.0f), self.m_direction); // une normale perpendiculaire a une norme vers le haut et la direction de la vue (droite dans l'espace vue)
31  vec3 up = cross(self.m_direction, right); // une normale perpendiculaire a la direction de la vue et du vecteur droite (haut dans l'espace vue)
32 
33  self.m_direction = rotate(self.m_direction, float(y - oldy) * -yangle, right);
34  self.m_direction = rotate(self.m_direction, float(x - oldx) * xangle, up);
35 
36  }
37 
38  oldx = x, oldy = y;
39 }

55 void Camera::update()
56 {
57  vec3 right = cross(vec3(0.0f, 1.0f, 0.0f), m_direction);
58  auto oldPos = m_position;
59 
60  // Avance ou recule dans la direction de la vue
61  if(glfwGetKey(*m_glWin, GLFW_KEY_UP) == GLFW_PRESS) m_position += m_direction * 0.1f;
62  if(glfwGetKey(*m_glWin, GLFW_KEY_DOWN) == GLFW_PRESS) m_position -= m_direction * 0.1f;
63  // Avance ou recule sur la droite de l'ecran
64  if(glfwGetKey(*m_glWin, GLFW_KEY_RIGHT) == GLFW_PRESS) m_position -= right * 0.1f;
65  if(glfwGetKey(*m_glWin, GLFW_KEY_LEFT) == GLFW_PRESS) m_position += right * 0.1f;
66 
67  vec2 mstick;
68  if(m_controller && length(mstick = m_controller->getMainStickPosition()) > 0.25f)
69  {
70  m_position += m_direction * -mstick.y * 0.1f;
71  m_position += right * -mstick.x * 0.1f;
72  }
73  if(m_controller && length(mstick = m_controller->getSecondaryStickPosition()) > 0.25f)
74  {
75  vec3 up = cross(m_direction, right);
76  m_direction = rotate(m_direction, mstick.y * 0.1f, right);
77  m_direction = rotate(m_direction, -mstick.x * 0.1f, up);
78  }
79  if(length(m_position) >= 9.0f) m_position = oldPos;
80  m_view = glm::lookAt(m_position,
81  m_direction + m_position,
82  glm::vec3(0.0f, 1.0f, 0.0f));
83 }
